001    package com.pusher.client;
002    
003    public class User {
004    
005        private final String id;
006        private final String jsonData;
007    
008        public User(String id, String jsonData) {
009            this.id = id;
010            this.jsonData = jsonData;
011        }
012    
013        public String getId() {
014            return id;
015        }
016    
017        public String getInfo() {
018            return jsonData;
019        }
020        
021        @Override
022        public String toString() {
023            return String.format("[User id=%s, data=%s]", id, jsonData);
024        }
025        
026        @Override
027        public int hashCode() {
028            return id.hashCode() + ((jsonData != null) ? jsonData.hashCode() : 0);
029        }
030        
031        @Override
032        public boolean equals(Object other) {
033            
034            if(other instanceof User) {
035                User otherUser = (User) other;
036                return this.getId().equals(otherUser.getId()) && this.getInfo().equals(otherUser.getInfo());
037            }
038            
039            return false;
040        }
041    }
