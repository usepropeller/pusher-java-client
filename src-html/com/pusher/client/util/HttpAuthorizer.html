001    package com.pusher.client.util;
002    
003    import java.io.BufferedReader;
004    import java.io.DataOutputStream;
005    import java.io.IOException;
006    import java.io.InputStream;
007    import java.io.InputStreamReader;
008    import java.net.HttpURLConnection;
009    import java.net.MalformedURLException;
010    import java.net.URL;
011    import java.net.URLEncoder;
012    import java.util.HashMap;
013    import com.pusher.client.AuthorizationFailureException;
014    import com.pusher.client.Authorizer;
015    
016    public class HttpAuthorizer implements Authorizer {
017    
018            private final URL endPoint;
019            private HashMap<String, String> mHeaders = new HashMap<String, String>();
020    
021            public HttpAuthorizer(String endPoint) {
022                    try {
023                            this.endPoint = Factory.newURL(endPoint);
024                    } catch (MalformedURLException e) {
025                            throw new IllegalArgumentException(
026                                            "Could not parse authentication end point into a valid URL", e);
027                    }
028            }
029            
030            public void setHeaders(HashMap<String, String> headers) {
031                    this.mHeaders = headers;
032            }
033    
034            @Override
035            public String authorize(String channelName, String socketId)
036                            throws AuthorizationFailureException {
037    
038                    try {
039                            String urlParameters = "channel_name="
040                                            + URLEncoder.encode(channelName, "UTF-8") + "&socket_id="
041                                            + URLEncoder.encode(socketId, "UTF-8");
042    
043                            HttpURLConnection connection = (HttpURLConnection) endPoint
044                                            .openConnection();
045                            connection.setDoOutput(true);
046                            connection.setDoInput(true);
047                            connection.setInstanceFollowRedirects(false);
048                            connection.setRequestMethod("POST");
049                            connection.setRequestProperty("Content-Type",
050                                            "application/x-www-form-urlencoded");
051                            connection.setRequestProperty("charset", "utf-8");
052                            connection.setRequestProperty("Content-Length",
053                                            "" + Integer.toString(urlParameters.getBytes().length));
054    
055                            // Add in the user defined headers
056                            for (String headerName : mHeaders.keySet()) {
057                                    String headerValue = mHeaders.get(headerName);
058                                    connection.setRequestProperty(headerName, headerValue);
059                            }
060    
061                            connection.setUseCaches(false);
062    
063                            // Send request
064                            DataOutputStream wr = new DataOutputStream(connection.getOutputStream());
065                            wr.writeBytes(urlParameters);
066                            wr.flush();
067                            wr.close();
068    
069                            // Read response
070                            InputStream is = connection.getInputStream();
071                            BufferedReader rd = new BufferedReader(new InputStreamReader(is));
072                            String line;
073                            StringBuffer response = new StringBuffer();
074                            while ((line = rd.readLine()) != null) {
075                                    response.append(line);
076                            }
077                            rd.close();
078    
079                            return response.toString();
080    
081                    } catch (IOException e) {
082                            throw new AuthorizationFailureException(e);
083                    }
084            }
085    }
