001    package com.pusher.client.channel;
002    
003    /**
004     * <p>Client applications should implement this interface if they want to be notified when events are received on a
005     * public or private channel.</p>
006     * 
007     * <p>To bind your implementation of this interface to a channel, either:
008     * <ul>
009     *  <li>Call {@link com.pusher.client.Pusher#subscribe(String)} to subscribe and receive an instance of {@link Channel}.</li>
010     *  <li>Call {@link Channel#bind(String, ChannelEventListener)} to bind your listener to a specified event.</li>
011     * </ul>
012     * 
013     * Or, call {@link com.pusher.client.Pusher#subscribe(String, ChannelEventListener, String...)} to subscribe to a channel and
014     * bind your listener to one or more events at the same time.</p>
015     *
016     */
017    public interface ChannelEventListener extends SubscriptionEventListener {
018    
019        /**
020         * <p>Callback that is fired when a subscription success acknowledgement message is received from Pusher after subscribing to
021         * the channel.</p>
022         * 
023         * <p>For public channels this callback will be more or less immediate, assuming that you are connected to Pusher at the time of
024         * subscription. For private channels this callback will not be fired unless you are successfully authenticated.</p>
025         * @param channelName The name of the channel that was successfully subscribed.
026         */
027        void onSubscriptionSucceeded(String channelName);
028    }
