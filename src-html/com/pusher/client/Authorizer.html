001    package com.pusher.client;
002    
003    /**
004     * Subscriptions to {@link com.pusher.client.channel.PrivateChannel Private} and {@link com.pusher.client.channel.PresenceChannel presence} channels
005     * need to be authorized. This interface provides an {@link #authorize} as a mechanism for doing this.
006     * 
007     * <p>See the {@link com.pusher.client.util.HttpAuthorizer HttpAuthorizer} as an example.</p>
008     */
009    public interface Authorizer {
010    
011                    /**
012                     * Called when a channel is to be authenticated.
013                     *
014                     * @param channelName The name of the channel to be authenticated.
015                     * @param socketId A unique socket connection ID to be used with the authentication.
016                     *      This uniquely identifies the connection that the subscription is being authenticated for.
017                     * @return An authentication token.
018                     * @throws AuthorizationFailureException if the authentication fails.
019                     */
020        String authorize(String channelName, String socketId) throws AuthorizationFailureException;
021    }
