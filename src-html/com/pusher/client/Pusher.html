001    package com.pusher.client;
002    
003    import com.pusher.client.channel.Channel;
004    import com.pusher.client.channel.ChannelEventListener;
005    import com.pusher.client.channel.PresenceChannel;
006    import com.pusher.client.channel.PresenceChannelEventListener;
007    import com.pusher.client.channel.PrivateChannel;
008    import com.pusher.client.channel.PrivateChannelEventListener;
009    import com.pusher.client.channel.impl.ChannelManager;
010    import com.pusher.client.channel.impl.InternalChannel;
011    import com.pusher.client.channel.impl.PresenceChannelImpl;
012    import com.pusher.client.channel.impl.PrivateChannelImpl;
013    import com.pusher.client.connection.Connection;
014    import com.pusher.client.connection.ConnectionEventListener;
015    import com.pusher.client.connection.ConnectionState;
016    import com.pusher.client.connection.impl.InternalConnection;
017    import com.pusher.client.util.Factory;
018    
019    public class Pusher {
020    
021            private final PusherOptions pusherOptions;
022            private final InternalConnection connection;
023            private final ChannelManager channelManager;
024    
025            /**
026             * <p>
027             * Creates a new instance of Pusher.
028             * </p>
029             * 
030             * <p>
031             * Note that if you use this constructor you will not be able to subscribe to private or presence channels because no {@link Authorizer} has been set. If you want to use private or presence channels:
032             * <ul>
033             * <li>Create an implementation of the {@link Authorizer} interface, or use the {@link com.pusher.client.util.HttpAuthorizer} provided.</li>
034             * <li>Create an instance of {@link PusherOptions} and set the authorizer on it by calling {@link PusherOptions#setAuthorizer(Authorizer)}.</li>
035             * <li>Use the {@link #Pusher(String, PusherOptions)} constructor to create an instance of Pusher.</li>
036             * </ul>
037             * </p>
038             * 
039             * <p>
040             * The {@link com.pusher.client.example.PrivateChannelExampleApp} and {@link com.pusher.client.example.PresenceChannelExampleApp} example applications show how to do this.
041             * </p>
042             * 
043             * @param apiKey Your Pusher API key.
044             */
045            public Pusher(String apiKey) {
046    
047                    this(apiKey, new PusherOptions());
048            }
049    
050            /**
051             * Creates a new instance of Pusher.
052             * 
053             * @param apiKey Your Pusher API key.
054             * @param pusherOptions Options for the Pusher client library to use.
055             */
056            public Pusher(String apiKey, PusherOptions pusherOptions) {
057                    
058                    if (apiKey == null || apiKey.isEmpty()) {
059                            throw new IllegalArgumentException("API Key cannot be null or empty");
060                    }
061    
062                    if (pusherOptions == null) {
063                            throw new IllegalArgumentException("PusherOptions cannot be null");
064                    }
065    
066                    this.pusherOptions = pusherOptions;
067                    this.connection = Factory.getConnection(apiKey, this.pusherOptions.isEncrypted());
068                    this.channelManager = Factory.getChannelManager();
069                    this.channelManager.setConnection(this.connection);
070            }
071    
072            /* Connection methods */
073    
074            /**
075             * Gets the underlying {@link Connection} object that is being used by this instance of {@linkplain Pusher}.
076             * 
077             * @return The {@link Connection} object.
078             */
079            public Connection getConnection() {
080                    return connection;
081            }
082    
083            /**
084             * Connects to Pusher. Any {@link ConnectionEventListener}s that have already been registered using the {@link Connection#bind(ConnectionState, ConnectionEventListener)} method will receive connection events.
085             */
086            public void connect() {
087                    connect(null);
088            }
089    
090            /**
091             * Binds a {@link ConnectionEventListener} to the specified events and then connects to Pusher. This is equivalent to binding a {@link ConnectionEventListener} using the {@link Connection#bind(ConnectionState, ConnectionEventListener)} method before connecting.
092             * 
093             * @param eventListener A {@link ConnectionEventListener} that will receive connection events. This can be null if you are not interested in receiving connection events, in which case you should call {@link #connect()} instead of this method.
094             * @param connectionStates An optional list of {@link ConnectionState}s to bind your {@link ConnectionEventListener} to before connecting to Pusher. If you do not specify any {@link ConnectionState}s then your {@link ConnectionEventListener} will be bound to all connection events. This is equivalent to calling {@link #connect(ConnectionEventListener, ConnectionState...)} with {@link ConnectionState#ALL}.
095             * @throws IllegalArgumentException If the {@link ConnectionEventListener} is null and at least one connection state has been specified.
096             */
097            public void connect(ConnectionEventListener eventListener, ConnectionState... connectionStates) {
098    
099                    if (eventListener != null) {
100                            if (connectionStates.length == 0) {
101                                    connectionStates = new ConnectionState[] { ConnectionState.ALL };
102                            }
103    
104                            for (ConnectionState state : connectionStates) {
105                                    connection.bind(state, eventListener);
106                            }
107                    } else {
108                            if (connectionStates.length > 0) {
109                                    throw new IllegalArgumentException("Cannot bind to connection states with a null connection event listener");
110                            }
111                    }
112    
113                    connection.connect();
114            }
115    
116            public void disconnect() {
117                    if (connection.getState() == ConnectionState.CONNECTED) {
118                            connection.disconnect();
119                    }
120            }
121    
122            /* Subscription methods */
123    
124            /**
125             * Subscribes to a public {@link Channel}.
126             * 
127             * @param channelName The name of the {@link Channel} to subscribe to.
128             * @return The {@link Channel} object representing your subscription.
129             */
130            public Channel subscribe(String channelName) {
131                    return subscribe(channelName, null);
132            }
133    
134            /**
135             * Binds a {@link ChannelEventListener} to the specified events and then subscribes to a public {@link Channel}.
136             * 
137             * @param channelName The name of the {@link Channel} to subscribe to.
138             * @param listener A {@link ChannelEventListener} to receive events. This can be null if you don't want to bind a listener at subscription time, in which case you should call {@link #subscribe(String)} instead of this method.
139             * @param eventNames An optional list of event names to bind your {@link ChannelEventListener} to before subscribing.
140             * @return The {@link Channel} object representing your subscription.
141             * @throws IllegalArgumentException If any of the following are true:
142             *             <ul>
143             *             <li>The channel name is null.</li>
144             *             <li>You are already subscribed to this channel.</li>
145             *             <li>The channel name starts with "private-". If you want to subscribe to a private channel, call {@link #subscribePrivate(String, PrivateChannelEventListener, String...)} instead of this method.</li>
146             *             <li>At least one of the specified event names is null.</li>
147             *             <li>You have specified at least one event name and your {@link ChannelEventListener} is null.</li>
148             *             </ul>
149             * @throws IllegalStateException If your {@link Connection} is not currently connected to Pusher.
150             */
151            public Channel subscribe(String channelName, ChannelEventListener listener, String... eventNames) {
152    
153                    InternalChannel channel = Factory.newPublicChannel(channelName);
154                    channelManager.subscribeTo(channel, listener, eventNames);
155    
156                    return channel;
157            }
158    
159            public PrivateChannel subscribePrivate(String channelName) {
160                    return subscribePrivate(channelName, null);
161            }
162    
163            public PrivateChannel subscribePrivate(String channelName, PrivateChannelEventListener listener, String... eventNames) {
164    
165                    throwExceptionIfNoAuthorizerHasBeenSet();
166    
167                    PrivateChannelImpl channel = Factory.newPrivateChannel(connection, channelName, pusherOptions.getAuthorizer());
168                    channelManager.subscribeTo(channel, listener, eventNames);
169    
170                    return channel;
171            }
172    
173            public PresenceChannel subscribePresence(String channelName) {
174                    return subscribePresence(channelName, null);
175            }
176    
177            public PresenceChannel subscribePresence(String channelName, PresenceChannelEventListener listener, String... eventNames) {
178    
179                    throwExceptionIfNoAuthorizerHasBeenSet();
180    
181                    PresenceChannelImpl channel = Factory.newPresenceChannel(connection, channelName, pusherOptions.getAuthorizer());
182                    channelManager.subscribeTo(channel, listener, eventNames);
183    
184                    return channel;
185            }
186    
187            public void unsubscribe(String channelName) {
188    
189                    if (connection.getState() != ConnectionState.CONNECTED) {
190                            throw new IllegalStateException("Cannot unsubscribe from channel " + channelName + " while not connected");
191                    }
192    
193                    channelManager.unsubscribeFrom(channelName);
194            }
195    
196            /* implementation detail */
197    
198            private void throwExceptionIfNoAuthorizerHasBeenSet() {
199                    if (pusherOptions.getAuthorizer() == null) {
200                            throw new IllegalStateException("Cannot subscribe to a private or presence channel because no Authorizer has been set. Call PusherOptions.setAuthorizer() before connecting to Pusher");
201                    }
202            }
203    }
