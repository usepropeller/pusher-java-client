001    package com.pusher.client.example;
002    
003    import java.util.Set;
004    
005    import com.pusher.client.Pusher;
006    import com.pusher.client.PusherOptions;
007    import com.pusher.client.channel.User;
008    import com.pusher.client.channel.PresenceChannel;
009    import com.pusher.client.channel.PresenceChannelEventListener;
010    import com.pusher.client.connection.ConnectionEventListener;
011    import com.pusher.client.connection.ConnectionStateChange;
012    import com.pusher.client.util.HttpAuthorizer;
013    
014    public class PresenceChannelExampleApp implements ConnectionEventListener,
015                    PresenceChannelEventListener {
016    
017            private final Pusher pusher;
018            private final String channelName;
019            private final String eventName;
020    
021            private PresenceChannel channel;
022    
023            public static void main(String[] args) {
024                    new PresenceChannelExampleApp(args);
025            }
026    
027            public PresenceChannelExampleApp(String[] args) {
028    
029                    String apiKey = (args.length > 0) ? args[0] : "a87fe72c6f36272aa4b1";
030                    channelName = (args.length > 1) ? args[1] : "presence-my-channel";
031                    eventName = (args.length > 2) ? args[2] : "my-event";
032    
033                    HttpAuthorizer authorizer = new HttpAuthorizer(
034                                    "http://www.leggetter.co.uk/pusher/pusher-examples/php/authentication/src/presence_auth.php");
035                    PusherOptions options = new PusherOptions().setAuthorizer(authorizer);
036    
037                    pusher = new Pusher(apiKey, options);
038                    pusher.connect(this);
039                    
040                    channel = pusher.subscribePresence(channelName, this, eventName);
041            }
042    
043            /* ConnectionEventListener implementation */
044    
045            @Override
046            public void onConnectionStateChange(ConnectionStateChange change) {
047    
048                    System.out.println(String.format(
049                                    "Connection state changed from [%s] to [%s]",
050                                    change.getPreviousState(), change.getCurrentState()));
051            }
052    
053            @Override
054            public void onError(String message, String code, Exception e) {
055    
056                    System.out.println(String.format(
057                                    "An error was received with message [%s], code [%s], exception [%s]",
058                                    message, code, e));
059            }
060    
061            /* PresenceChannelEventListener implementation */
062    
063            @Override
064            public void onUsersInformationReceived(String channelName, Set<User> users) {
065    
066                    System.out.println("Received user information");
067    
068                    printCurrentlySubscribedUsers();
069            }
070    
071            @Override
072            public void userSubscribed(String channelName, User user) {
073    
074                    System.out.println(String.format("A new user has joined channel [%s]: %s",
075                                    channelName, user.toString()));
076    
077                    printCurrentlySubscribedUsers();
078            }
079    
080            @Override
081            public void userUnsubscribed(String channelName, User user) {
082    
083                    System.out.println(String.format("A user has left channel [%s]: %s",
084                                    channelName, user));
085    
086                    printCurrentlySubscribedUsers();
087            }
088    
089            @Override
090            public void onEvent(String channelName, String eventName, String data) {
091    
092                    System.out.println(String.format(
093                                    "Received event [%s] on channel [%s] with data [%s]", eventName,
094                                    channelName, data));
095            }
096    
097            @Override
098            public void onSubscriptionSucceeded(String channelName) {
099    
100                    System.out.println(String.format("Subscription to channel [%s] succeeded",
101                                    channel.getName()));
102            }
103    
104            @Override
105            public void onAuthenticationFailure(String message, Exception e) {
106    
107                    System.out.println(String.format(
108                                    "Authentication failure due to [%s], exception was [%s]", message, e));
109            }
110    
111            private void printCurrentlySubscribedUsers() {
112                    StringBuilder sb = new StringBuilder("Users now subscribed to the channel:");
113                    for (User remainingUser : channel.getUsers()) {
114                            sb.append("\n\t");
115                            sb.append(remainingUser.toString());
116    
117                            if (remainingUser.equals(channel.getMe())) {
118                                    sb.append(" (me)");
119                            }
120                    }
121    
122                    System.out.println(sb.toString());
123            }
124    }
